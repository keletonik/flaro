pythonimport requests

AIRTABLE_TOKEN = "your_token_here"
BASE_ID = "appuKqojpI3bmO79D"

TABLES = {
    "jobs":     "tblfX6SQYVJJmkdeK",
    "quotes":   "tbl6k5nXbDVGE4Pnd",
    "contacts": "tblzyaBUUZTF8fn6J"
}

HEADERS = {
    "Authorization": f"Bearer {AIRTABLE_TOKEN}",
    "Content-Type": "application/json"
}

def get_records(table_name):
    url = f"https://api.airtable.com/v0/{BASE_ID}/{TABLES[table_name]}"
    records, offset = [], None
    while True:
        params = {"offset": offset} if offset else {}
        r = requests.get(url, headers=HEADERS, params=params)
        data = r.json()
        records.extend(data.get("records", []))
        offset = data.get("offset")
        if not offset:
            break
    return records

def update_record(table_name, record_id, fields):
    url = f"https://api.airtable.com/v0/{BASE_ID}/{TABLES[table_name]}/{record_id}"
    return requests.patch(url, headers=HEADERS, json={"fields": fields}).json()

def create_record(table_name, fields):
    url = f"https://api.airtable.com/v0/{BASE_ID}/{TABLES[table_name]}"
    return requests.post(url, headers=HEADERS, json={"fields": fields}).json()

# Example usage
jobs = get_records("jobs")
print(f"Loaded {len(jobs)} jobs")
